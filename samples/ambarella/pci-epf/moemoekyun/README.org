#!/bin/sh
USCOUNTER="1"
TEN="10000"

while [ $USCOUNTER -le $TEN ];
do
   echo "load count: $USCOUNTER"
   modprobe $1;
   rmmod $1;
   USCOUNTER=$(expr $USCOUNTER + 1)
done
